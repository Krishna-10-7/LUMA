React Code
import React, { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);
  
  return (
    <div>
      <h1>{count}</h1>
      <button onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
}
LumaJS Code
const { h, useState } = Luma;

function Counter() {
  const [count, setCount] = useState(0);
  
  return h('div', {},
    h('h1', {}, count),
    h('button', { 
      onClick: () => setCount(count + 1) 
    }, 'Increment')
  );
}

Same API, No JSX needed! LumaJS provides a React-compatible API with hooks, but without the build step. Use h() directly or add JSX support with Babel/ESBuild.
